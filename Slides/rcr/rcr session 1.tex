\begin{frame}
	\frametitle{Kinds of IRB Review}
	
	\begin{itemize}
		\item Exempt review
		\begin{itemize}
			\item This is available for some kinds of low-risk surveys, interviews, and observation-based research
			\item It is also available for some kinds of low-risk experiments, involving benign behavioral interventions in conjunction with the collection of information from an adult subject through verbal or written responses (including data entry) or audiovisual recording if the subject prospectively agrees to the intervention and information collection
		\end{itemize}
	\end{itemize}
	
\end{frame}

\begin{frame}
	\frametitle{Kinds of IRB Review}
	
	\begin{itemize}
		\item Expedited review
		\begin{itemize}
			\item Collection of data from voice, video, digital, or image recordings made for research purposes.
			
		\end{itemize}
	\end{itemize}
	
\end{frame}

\begin{frame}
	\frametitle{Kinds of IRB Review}
	
	\begin{itemize}
		\item Expedited review
		\begin{itemize}
			\item Research on individual or group characteristics or behavior (including, but not limited to, research on perception, cognition, motivation, identity, language, communication, cultural beliefs or practices, and social behavior) or research employing survey, interview, oral history, focus group, program evaluation, human factors evaluation, or quality assurance methodologies.
		\end{itemize}
	\end{itemize}
	
\end{frame}
